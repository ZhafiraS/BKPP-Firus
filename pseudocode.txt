Program_TarifParkir
Kamus
    tipeKendaraan : string
    jam1Motor, jam2Motor, jam1Mobil, jam2Mobil, jam1Sepeda, jam2Sepeda, jam1Bus, jam2Bus : integer
    jam, menit, tarif, pilihan, pembayaran : integer
    stop : boolean
Algoritma
    jam1Motor = 3000
    jam2Motor = 2000
    jam1Mobil = 5000
    jam2Mobil = 3000
    jam1Sepeda = 1000
    jam2Sepeda = 500
    jam1Bus = 10000
    jam2Bus = 7000
    tarif = 0

    repeat
        output("Masukkan tipe kendaraan (Motor/Mobil/Sepeda/Bus): ")
        repeat
            input(tipeKendaraan)
            while tipeKendaraan != "Motor" AND tipeKendaraan != "Mobil" AND tipeKendaraan != "Sepeda" AND tipeKendaraan != "Bus" then
                output("Input invalid, kendaraan hanya Motor/Mobil/Sepeda/Bus")
                output("Masukkan tipe kendaraan: ")
            endwhile
        until (tipeKendaraan == "Motor" OR tipeKendaraan == "Mobil" OR tipeKendaraan == "Sepeda" OR tipeKendaraan == "Bus")

        output("Masukkan jam parkir: ")
        repeat
            input(jam)
            if jam < 0 then
                output("Input invalid, jam hanya bisa bernilai 0 atau lebih)
                output("Masukkan jam parkir")
            endif
        until (jam >= 0)

        output("Masukkan menit parkir: ")
        repeat
            input(menit)
            if menit < 0 OR menit > 59 then
                output("Input invalid, menit hanya bisa bernilai 0-59")
                output("Masukkan menit parkir: ")
            endif
        until (menit > 0 AND menit < 59)

        if jam == 0 AND menit == 0 then
            output("Kendaraan tidak parkir")
        else
            if tipeKendaraan = "Motor" then
                if jam == 0 AND menit != 0 then
                    tarif ← jam1Motor
                else 
                    tarif ← ((jam - 1)* jam2Motor) + jam1Motor
                    if menit != 0 then
                        tarif ← tarif + jam2Motor
                    endif
                endif
            else if tipeKendaraan = "Mobil" then
                if jam == 0 AND menit != 0 then
                    tarif ← jam1Mobil
                else
                    tarif ← ((jam - 1) * jam2Mobil) + jam1Mobil
                    if menit != 0 then
                        tarif ← tarif + jam2Mobil
                    endif
                endif
            else if tipeKendaraan = "Sepeda" then
                if jam == 0 AND menit != 0 then
                    tarif ← jam1Sepeda
                else
                    tarif ← ((jam - 1) * jam2Sepeda) + jam1Sepeda
                    if menit != 0 then
                        tarif ← tarif + jam2Sepeda
                    endif
                endif
            else if tipeKendaraan = "Bus" then
                if jam == 0 AND menit != 0 then
                    tarif ← jam1Bus
                else
                    tarif ← ((jam - 1) * jam2Bus) + jam1Bus
                    if menit != 0 then
                        tarif ← tarif + jam2Bus
                    endif
                endif
            endif
        endif
        output("Tarif parkir yang harus dibayar adalah Rp.", tarif, " dengan tipe kendaraan", tipeKendaraan")
        output("Metode Pembayaran: ")
        output("1. QRIS")
        output("2. Cash")
        output("Pilihan (1/2): ")
        repeat
            input(pembayaran)
            if pembayaran != 1 AND pembayaran != 2 then
                output("Pilihan tidak valid, masukkan 1 atau 2")
                output("Pilihan (1/2): ")
            endif
        until (pembayaran == 1 OR pilihan == 2)
                

        output("Apakah ingin memasukkan tarif parkir lagi?")
        output("1. Iya")
        output("2. Tidak")
        output("Pilihan (1/2): ")
    
        repeat
            input(pilihan)
            if pilihan != 1 AND pilihan != 2 then
                output("Pilihan tidak valid, masukkan 1 atau 2")
                output("Pilihan (1/2): ")
            endif
        until (pilihan == 1 OR pilihan == 2)

    until(pilihan == 2)
    output("Program Selesai")

Endprogram
